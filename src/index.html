Makefile 示例

基础 C 项目 简单

# 简单C项目示例
CC = gcc
CFLAGS = -Wall -Wextra
LDFLAGS = 

all: myapp

myapp: main.o utils.o
    $(CC) -o $@ $^ $(CFLAGS) $(LDFLAGS)
    
main.o: main.c utils.h
    $(CC) -c $< $(CFLAGS)
    
utils.o: utils.c utils.h
    $(CC) -c $< $(CFLAGS)
    
clean:
    rm -f *.o myapp
.PHONY: clean

C++ 项目 中级

# C++项目示例
CXX = g++
CXXFLAGS = -std=c++11 -Wall -O2
LDFLAGS = 

all: application

application: main.o module.o
    $(CXX) -o $@ $^ $(CXXFLAGS) $(LDFLAGS)
    
main.o: main.cpp module.h
    $(CXX) -c $< $(CXXFLAGS)
    
module.o: module.cpp module.h
    $(CXX) -c $< $(CXXFLAGS)
    
clean:
    rm -f *.o application
.PHONY: clean all
